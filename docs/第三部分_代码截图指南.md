### 4. 代码截图：语言检测算法

```python
def detect_language(self, text: str) -> Tuple[str, float]:
    """检测文本语言
    返回: (语言代码, 置信度)
    支持: zh-中文, en-英文, es-西班牙语, fr-法语, de-德语, ja-日语, ko-韩语
    """
    # 简单的语言检测逻辑
    chinese_chars = len(re.findall(r'[\u4e00-\u9fff]', text))
    total_chars = len(text.strip())
    
    if total_chars == 0:
        return 'unknown', 0.0
    
    chinese_ratio = chinese_chars / total_chars
    
    # 如果中文字符超过30%，认为是中文
    if chinese_ratio > 0.3:
        return 'zh', chinese_ratio
    
    # 检测其他语言的常见词
    text_lower = text.lower()
    
    # 英语常见词
    english_words = ['the', 'is', 'are', 'what', 'how', 'why', 'who', 'when', 'where']
    english_count = sum(1 for word in english_words if word in text_lower)
    
    # 西班牙语常见词
    spanish_words = ['qué', 'cómo', 'por qué', 'quién', 'cuándo', 'dónde', 'es', 'son']
    spanish_count = sum(1 for word in spanish_words if word in text_lower)
    
    if english_count >= 2:
        return 'en', 0.8
    elif spanish_count >= 1:
        return 'es', 0.7
    
    return 'en', 0.5  # 默认返回英文
```

**图注**：语言检测算法通过字符比例和关键词匹配实现多语言识别。

```python
    """分析古诗中的字词"""

        user_id = get_jwt_identity()

            question=question_text,
            answer=result['content'],
```

```python
def evaluate_pronunciation(self, user_audio_path: str, reference_text: str) -> Dict[str, Any]:
    """评估发音质量
    
    Args:
        user_audio_path: 用户录音文件路径
        reference_text: 参考文本
    
    Returns:
        {
            'overall_score': int,      # 总分 (0-100)
            'stars': int,              # 星级 (1-5)
            'accuracy_score': int,     # 准确度分数
            'fluency_score': int,      # 流利度分数
            'tone_score': int,         # 音调分数
            'transcribed_text': str,   # 识别的文本
            'word_errors': list,       # 错误字词列表
            'feedback': str            # 反馈建议
        }
    """
    try:
        # 1. 使用Whisper识别用户朗读
        transcription = self.transcribe_audio(user_audio_path)
        user_text = transcription['text']
        
        # 2. 计算准确度分数
        accuracy_score, word_errors = self._calculate_accuracy(reference_text, user_text)
        
        # 3. 计算流利度分数
        fluency_score = self._calculate_fluency(user_audio_path, transcription)
        
        # 4. 计算音调分数
        tone_score = self._calculate_tone(user_audio_path)
        
        # 5. 综合评分（加权平均）
        overall_score = int(
            accuracy_score * 0.5 +  # 准确度权重50%
            fluency_score * 0.3 +   # 流利度权重30%
            tone_score * 0.2        # 音调权重20%
        )
        
        # 6. 星级评定
        if overall_score >= 90:
            stars = 5
        elif overall_score >= 80:
            stars = 4
        elif overall_score >= 70:
            stars = 3
        elif overall_score >= 60:
            stars = 2
        else:
            stars = 1
        
        return {
            'overall_score': overall_score,
            'stars': stars,
            'accuracy_score': accuracy_score,
            'fluency_score': fluency_score,
            'tone_score': tone_score,
            'transcribed_text': user_text,
            'word_errors': word_errors,
            'feedback': self._generate_feedback(overall_score, word_errors)
        }
```

**图注**：图 3-4 展示了发音评估的核心算法，通过 Whisper 识别、多维度分析和加权计算得出综合评分。

```python
async def generate_tts_audio(
    self, 
    text: str, 
    voice: str = 'female_standard',
    rate: str = 'normal',
    output_path: str = None
) -> str:
    """生成TTS音频
    
    Args:
        text: 要朗读的文本
        voice: 音色选择
        rate: 语速
        output_path: 输出文件路径
    
    Returns:
        音频文件路径
    """
    if output_path is None:
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        output_path = f"static/audio/tts_{timestamp}.mp3"
    
    # 确保目录存在
    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    
    # 获取音色和语速
    voice_name = self.tts_voices.get(voice, self.tts_voices['female_standard'])
    rate_value = self.speech_rates.get(rate, self.speech_rates['normal'])
    
    # 生成音频
    communicate = edge_tts.Communicate(text, voice_name, rate=rate_value)
    await communicate.save(output_path)
    
    print(f"✓ TTS音频已生成: {output_path}")
    return output_path
```

```python
def _calculate_accuracy(self, reference: str, user_text: str) -> Tuple[int, List[Dict]]:
    """计算准确度分数和错误字词
    
    Returns:
        (准确度分数, 错误字词列表)
    """
    # 清理文本
    ref_clean = self._clean_text(reference)
    user_clean = self._clean_text(user_text)
    
    # 使用difflib进行对比
    matcher = difflib.SequenceMatcher(None, ref_clean, user_clean)
    similarity = matcher.ratio()
    
    # 找出错误的字词
    word_errors = []
    opcodes = matcher.get_opcodes()
    
    for tag, i1, i2, j1, j2 in opcodes:
        if tag == 'replace':
            word_errors.append({
                'type': 'replace',
                'expected': ref_clean[i1:i2],
                'actual': user_clean[j1:j2],
                'position': i1
            })
        elif tag == 'delete':
            word_errors.append({
                'type': 'missing',
                'expected': ref_clean[i1:i2],
                'actual': '',
                'position': i1
            })
        elif tag == 'insert':
            word_errors.append({
                'type': 'extra',
                'expected': '',
                'actual': user_clean[j1:j2],
                'position': i1
            })
    
    accuracy_score = int(similarity * 100)
    return accuracy_score, word_errors
```

**图注**：准确度计算使用 difflib 库进行文本对比，识别替换、遗漏和多余的字词。
