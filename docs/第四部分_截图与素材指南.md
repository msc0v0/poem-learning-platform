```javascript
// 播放意境视频
function playVideo() {
    if (!currentPoem || !currentPoem.title) {
        alert('请先加载古诗');
        return;
    }
    
    // 根据古诗标题查找对应的视频文件
    const videoFileName = `${currentPoem.title}.mp4`;
    const videoPath = `/static/videos/${videoFileName}`;
    
    // 设置视频源
    const videoSource = document.getElementById('videoSource');
    const videoElement = document.getElementById('poemVideo');
    const videoContainer = document.getElementById('videoPlayerContainer');
    
    videoSource.src = videoPath;
    videoElement.load();
    
    // 添加视频事件监听器
    if (!videoElement.dataset.trackingEnabled) {
        videoElement.addEventListener('play', () => trackVideoPlay(videoElement));
        videoElement.addEventListener('pause', () => trackVideoPause(videoElement));
        videoElement.addEventListener('seeked', () => trackVideoSeek(videoElement));
        videoElement.addEventListener('ended', () => trackVideoEnded(videoElement));
        videoElement.dataset.trackingEnabled = 'true';
    }
    
    // 显示视频容器
    videoContainer.style.display = 'block';
    videoContainer.classList.add('video-fade-in');
    
    // 滚动到视频位置
    videoContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });
    
    // 视频加载完成后自动播放
    videoElement.addEventListener('loadeddata', function() {
        videoElement.play();
    }, { once: true });
}
```

**图注**：前端视频播放逻辑，包括视频源设置、事件监听和自动播放功能。

```javascript
// 视频播放追踪
function trackVideoPlay(videoElement) {
    videoWatchStart = Date.now();
    trackEvent('video_play', 'poem_video', {
        currentTime: videoElement.currentTime,
        duration: videoElement.duration,
        volume: videoElement.volume
    }, false);
    
    // 更新会话的视频数据
    if (currentSessionId) {
        fetch(`${API_BASE}/tracking/session/${currentSessionId}/video-engagement`, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${authToken}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                playback_event: {
                    type: 'play',
                    timestamp: new Date().toISOString(),
                    currentTime: videoElement.currentTime
                }
            })
        }).catch(err => console.error('视频追踪失败:', err));
    }
}

function trackVideoPause(videoElement) {
    const watchDuration = videoWatchStart ? (Date.now() - videoWatchStart) / 1000 : 0;
    
    trackEvent('video_pause', 'poem_video', {
        currentTime: videoElement.currentTime,
        duration: videoElement.duration,
        watchDuration: watchDuration
    }, false);
    
    if (currentSessionId) {
        fetch(`${API_BASE}/tracking/session/${currentSessionId}/video-engagement`, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${authToken}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                playback_event: {
                    type: 'pause',
                    timestamp: new Date().toISOString(),
                    currentTime: videoElement.currentTime
                },
                watch_segment: {
                    start: Math.max(0, videoElement.currentTime - watchDuration),
                    end: videoElement.currentTime,
                    duration: watchDuration
                }
            })
        }).catch(err => console.error('视频追踪失败:', err));
    }
    
    videoWatchStart = null;
}
```

**图注**：视频观看行为追踪代码，记录播放、暂停、快进等操作，用于学习分析。

```python
from datetime import datetime
from . import db

class Poem(db.Model):
    __tablename__ = 'poems'
    
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    author = db.Column(db.String(50), nullable=False)
    dynasty = db.Column(db.String(20), nullable=False)
    content = db.Column(db.Text, nullable=False)
    translation = db.Column(db.Text)  # 现代文翻译
    background = db.Column(db.Text)   # 创作背景
    analysis = db.Column(db.Text)     # 诗歌赏析
    english_notes = db.Column(db.Text)  # 英文注释
    
    # 多媒体资源
    video_path = db.Column(db.String(200))  # 意境视频路径
    audio_path = db.Column(db.String(200))  # TTS音频路径
    
    # 字词注释 (JSON格式存储)
    annotations = db.Column(db.Text)
    
    # 难度等级 (1-5)
    difficulty_level = db.Column(db.Integer, default=1)
    
    # 标签
    tags = db.Column(db.String(200))
    
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    
    def to_dict(self):
        return {
            'id': self.id,
            'title': self.title,
            'author': self.author,
            'dynasty': self.dynasty,
            'content': self.content,
            'translation': self.translation,
            'background': self.background,
            'analysis': self.analysis,
            'english_notes': self.english_notes,
            'video_path': self.video_path,
            'audio_path': self.audio_path,
            'annotations': self.annotations,
            'difficulty_level': self.difficulty_level,
            'tags': self.tags.split(',') if self.tags else [],
            'created_at': self.created_at.isoformat()
        }
```

```python
import requests

def generate_script_with_doubao(poem_data, hsk_level='HSK3'):
    """使用豆包大模型生成古诗讲稿"""
    
    # 豆包 API 配置
    api_url = "https://api.doubao.com/v1/chat/completions"
    api_key = os.getenv('DOUBAO_API_KEY')
    
    # 构建提示词
    system_prompt = f"""你是一位专业的古诗词教师，专门为国际汉语学习者讲解中国古诗。
请为以下古诗生成一份教学讲稿，要求：
1. 语言难度：适配 {hsk_level} 水平
2. 讲解结构：开场白、逐句解析、意境赏析、文化拓展、结束语
3. 语言风格：亲切自然、深入浅出、富有感染力
4. 时长控制：总时长约 4-5 分钟
5. 多语言：关键术语提供英文注释"""
    
    user_prompt = f"""古诗信息：
标题：{poem_data['title']}
作者：{poem_data['author']}
朝代：{poem_data['dynasty']}
内容：{poem_data['content']}

请生成完整的讲稿，使用口语化的表达，适合数字人播报。"""
    
    # 调用豆包 API
    response = requests.post(
        api_url,
        headers={
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        },
        json={
            "model": "doubao-pro",
            "messages": [
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt}
            ],
            "temperature": 0.7,
            "max_tokens": 2000
        }
    )
    
    result = response.json()
    script = result['choices'][0]['message']['content']
    
    return script
```

### 8. 示意图：SeedDance API 调用流程（伪代码）

**代码示例**：
```python
import requests
import asyncio

async def generate_video_with_seeddance(script, avatar_id='teacher_female_01'):
    """使用即梦 SeedDance 2.0 生成数字人视频"""
    
    # SeedDance API 配置
    api_url = "https://api.seeddance.com/v2/generate"
    api_key = os.getenv('SEEDDANCE_API_KEY')
    
    # 将讲稿分段
    script_segments = split_script_into_segments(script)
    
    # 构建请求参数
    payload = {
        "avatar_id": avatar_id,  # 数字人形象ID
        "script": script_segments,
        "voice": {
            "speaker": "zh-CN-XiaoxiaoNeural",  # 音色
            "speed": 1.0,  # 语速
            "pitch": 0,  # 音调
            "volume": 1.0  # 音量
        },
        "background": {
            "type": "image",
            "url": "/static/backgrounds/classical_chinese.jpg"
        },
        "subtitle": {
            "enabled": True,
            "languages": ["zh", "zh-pinyin", "en"]
        },
        "resolution": "1080p",
        "format": "mp4"
    }
    
    # 调用 SeedDance API
    response = requests.post(
        api_url,
        headers={"Authorization": f"Bearer {api_key}"},
        json=payload
    )
    
    result = response.json()
    
    # 等待视频生成完成
    task_id = result['task_id']
    video_url = await poll_video_generation_status(task_id)
    
    return video_url

async def poll_video_generation_status(task_id):
    """轮询视频生成状态"""
    status_url = f"https://api.seeddance.com/v2/task/{task_id}"
    
    while True:
        response = requests.get(status_url)
        result = response.json()
        
        if result['status'] == 'completed':
            return result['video_url']
        elif result['status'] == 'failed':
            raise Exception(f"视频生成失败: {result['error']}")
        
        await asyncio.sleep(5)  # 每5秒查询一次
```

**图注**：图 4-5 展示了如何调用 SeedDance API 生成数字人讲解视频。
